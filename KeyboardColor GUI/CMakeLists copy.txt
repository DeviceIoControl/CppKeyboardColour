cmake_minimum_required(VERSION 3.16)

project(KeyboardColor VERSION 0.1 LANGUAGES CXX)

set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(QT NAMES Qt6 Qt5 REQUIRED COMPONENTS Widgets)
find_package(Qt${QT_VERSION_MAJOR} REQUIRED COMPONENTS Widgets)

set(PROJECT_SOURCES
        main.cpp
        keyboardcoulour.cpp
        keyboardcoulour.h
        keyboardcoulour.ui
)

if(${QT_VERSION_MAJOR} GREATER_EQUAL 6)
    qt_add_executable(KeyboardColor
        MANUAL_FINALIZATION
        ${PROJECT_SOURCES}
        resources.qrc
    )
# Define target properties for Android with Qt 6 as:
#    set_property(TARGET KeyboardColor APPEND PROPERTY QT_ANDROID_PACKAGE_SOURCE_DIR
#                 ${CMAKE_CURRENT_SOURCE_DIR}/android)
# For more information, see https://doc.qt.io/qt-6/qt-add-executable.html#target-creation
else()
    if(ANDROID)
        add_library(KeyboardColor SHARED
            ${PROJECT_SOURCES}
        )
# Define properties for Android with Qt 5 after find_package() calls as:
#    set(ANDROID_PACKAGE_SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/android")
    else()
        add_executable(KeyboardColor
            ${PROJECT_SOURCES}
        )
    endif()
endif()
target_link_options(KeyboardColor PRIVATE -Wl,--allow-multiple-definition)
target_link_libraries(KeyboardColor PRIVATE Qt${QT_VERSION_MAJOR}::Widgets)

# Qt for iOS sets MACOSX_BUNDLE_GUI_IDENTIFIER automatically since Qt 6.1.
# If you are developing for iOS or macOS you should consider setting an
# explicit, fixed bundle identifier manually though.
if(${QT_VERSION} VERSION_LESS 6.1.0)
  set(BUNDLE_ID_OPTION MACOSX_BUNDLE_GUI_IDENTIFIER com.example.KeyboardColor)
endif()
set_target_properties(KeyboardColor PROPERTIES
    ${BUNDLE_ID_OPTION}
    MACOSX_BUNDLE_BUNDLE_VERSION ${PROJECT_VERSION}
    MACOSX_BUNDLE_SHORT_VERSION_STRING ${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}
    MACOSX_BUNDLE TRUE
    WIN32_EXECUTABLE TRUE
)

include(GNUInstallDirs)
install(TARGETS KeyboardColor
    BUNDLE DESTINATION .
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)
set(ICON_FILE_NAME "votre_logo.ico") # Pour Windows
set(ICNS_FILE_NAME "votre_logo.icns") # Pour macOS

if(WIN32)
    set(APP_ICON_RC_FILE "${CMAKE_CURRENT_SOURCE_DIR}/appicon.rc")
    add_custom_command(
        OUTPUT ${APP_ICON_RC_FILE}
        COMMAND ${CMAKE_COMMAND} -E copy_if_different ${CMAKE_CURRENT_SOURCE_DIR}/${ICON_FILE_NAME} ${APP_ICON_RC_FILE}
        COMMENT "Copying icon file for Windows resource"
    )
    set_target_properties(KeyboardColor PROPERTIES
        WIN32_EXECUTABLE ON
        WIN32_ICON "${ICON_FILE_NAME}"
        RC_FILE "${APP_ICON_RC_FILE}"
    )
endif()

if(APPLE)
    set_target_properties(KeyboardColor PROPERTIES MACOSX_BUNDLE_ICON_FILE "${ICNS_FILE_NAME}")
endif()
if(QT_VERSION_MAJOR EQUAL 6)
    qt_finalize_executable(KeyboardColor)
endif()
